-- Créer la base de données
CREATE DATABASE Sportify;

-- Utiliser la base de données Sportify
USE Sportify;

-- Créer les tables
CREATE TABLE Utilisateurs (
    utilisateur_id INT PRIMARY KEY AUTO_INCREMENT,
    nom VARCHAR(100),
    email VARCHAR(100) UNIQUE,
    mot_de_passe VARCHAR(255),
    type_utilisateur ENUM('client', 'coach', 'admin'),
    adresse VARCHAR(255),
    numero_telephone VARCHAR(20),
    carte_etudiant VARCHAR(50)
);

CREATE TABLE Coachs (
    coach_id INT PRIMARY KEY AUTO_INCREMENT,
    utilisateur_id INT,
    specialite VARCHAR(100),
    disponibilite TEXT,
    bio TEXT,
    photo VARCHAR(255),
    FOREIGN KEY (utilisateur_id) REFERENCES Utilisateurs(utilisateur_id)
);

CREATE TABLE RendezVous (
    rendezvous_id INT PRIMARY KEY AUTO_INCREMENT,
    utilisateur_id INT,
    coach_id INT,
    date_rendezvous DATETIME,
    statut ENUM('confirmé', 'annulé'),
    FOREIGN KEY (utilisateur_id) REFERENCES Utilisateurs(utilisateur_id),
    FOREIGN KEY (coach_id) REFERENCES Coachs(coach_id)
);

CREATE TABLE InstallationsSportives (
    installation_id INT PRIMARY KEY AUTO_INCREMENT,
    nom VARCHAR(100),
    description TEXT,
    regles TEXT,
    disponibilite TEXT
);
